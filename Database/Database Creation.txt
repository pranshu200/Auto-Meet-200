#Create the following Tables.
-> User_Table
>create table User_Table(
	User_Id int Primary Key check(user_Id>0), 
	User_Name varchar(50) not null, 
	User_Email varchar(100) not null,
	User_Phone varchar(10) not null default 0, 
	User_Role Varchar(30) not null, 
	User_Credit int not null default 0 check(User_Credit<2001)
);

->Meeting_Room
>create table Meeting_Room(
	Room_Id int primary key,
	Room_Name varchar(50) not null unique,
	Room_Capacity int not null check(Room_Capacity>0),
	Room_Cost int not null,
	Room_Rating int not null
);

->Room_Amenities
>create table Room_Amenities(
	Room_Id int not null, 
	Amenity varchar(30) not null
);

->Room_Type
>create table Room_Type(
	Room_Id int not null,
	Room_Type varchar(50) not null,
	primary key(Room_Id,Room_Type),
	foreign key(Room_Id) references Meeting_Room(Room_Id)
);

->Booking_Info
>create table Booking_Info(
	Booking_Id int primary key,
	Room_Id int not null,
	User_Id int not null,
	Booking_Date date not null,
	Start_Time time not null,
	End_Time time not null,
	foreign key(Room_Id) references Meeting_Room(Room_Id),
	foreign key(User_Id) references User_Table(User_Id)
); 

->Room_Members
>create table Room_Members(
	Room_Id int not null,
	User_Id int not null,
	primary key(Room_Id, User_Id),
	foreign key(Room_Id) references Meeting_Room(Room_Id),
	foreign key(User_Id) references User_Table(User_Id)
);
 
->Meeting
>create table Meeting(
	Meeting_Id int not null,
	Meeting_Title varchar(50) not null,
	Meeting_Type varchar(50) not null,
	Booking_Id int not null,
	Primary Key(Meeting_Id,Booking_Id),
	Foreign Key(Booking_Id) references Booking_Info(Booking_Id)
);
